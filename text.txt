      const [story, setStory] = useState([]);

      const [timeLeft, setTimeLeft] = useState(30);
      const [clickedEmoji, setClickedEmoji] = useState(null);

      const emojis = ["🪼","😏","🚀","✨","🎯","💸","💪","🤣","🐙","👀","🔥","🌈","🍕","🤖","🌊"];
